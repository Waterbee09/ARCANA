function typeText(element, text, speed = 100) {
  element.innerHTML = '';
  let i = 0;
  const interval = setInterval(() => {
    if (text.charAt(i) === '\n') element.innerHTML += '<br>';
    else element.innerHTML += text.charAt(i);
    i++;
    if (i >= text.length) clearInterval(interval);
  }, speed);
}
